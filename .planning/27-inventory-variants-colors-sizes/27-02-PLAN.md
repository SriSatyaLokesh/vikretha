      for (const sizeRow of sizeSubRows) {

        variants.push({ color: colorName, size: sizeName, qty: sizeQty });
      }
    }
  } else {
    for (const colorRow of colorRows) {
      const colorName = colorRow.querySelector('.inv-color-name-input').value.trim();
      const colorQty  = parseInt(colorRow.querySelector('.inv-color-qty-input').value, 10);

      variants.push({ color: colorName, qty: colorQty });
    }

  // Compute aggregate stock from variants

  hasSizes = false; // has_colors replaces old hasSizes mode

```js
function _showVariantPicker(container, inv) {
  // Build a bottom-sheet showing color options (and sizes if has_sizes)
  document.body.style.overflow = 'hidden';
  const overlay = document.createElement('div');
  overlay.style.cssText = 'position:fixed;inset:0;background:rgba(0,0,0,0.4);z-index:60;display:flex;align-items:flex-end;';
  const sheet = document.createElement('div');
  sheet.style.cssText = 'background:var(--bg-primary);border-radius:16px 16px 0 0;padding:20px 16px;width:100%;max-height:75vh;overflow-y:auto;box-sizing:border-box;';
  const title = document.createElement('h3');
  title.style.cssText = 'font-size:1rem;font-weight:700;color:var(--text-primary);margin:0 0 16px;';
  title.textContent = inv.name + ' — Select Variant';
  sheet.appendChild(title);

  if (inv.has_sizes) {
    // Group variants by color
    const colorGroups = {};
    inv.variants.forEach(v => {
      if (!colorGroups[v.color]) colorGroups[v.color] = [];
      colorGroups[v.color].push(v);
    });
    Object.entries(colorGroups).forEach(([color, variants]) => {
      const colorLabel = document.createElement('div');
      colorLabel.style.cssText = 'font-size:0.82rem;font-weight:600;color:var(--text-secondary);margin-bottom:6px;';
      colorLabel.textContent = color;
      sheet.appendChild(colorLabel);
      const sizesRow = document.createElement('div');
      sizesRow.style.cssText = 'display:flex;flex-wrap:wrap;gap:8px;margin-bottom:14px;';
      variants.forEach(v => {
        const btn = document.createElement('button');
        btn.type = 'button';
        btn.textContent = v.size + (v.qty === 0 ? ' (out)' : '');
        btn.disabled = v.qty === 0;
        btn.className = 'btn btn-secondary btn-sm';
        btn.addEventListener('click', () => {
          const cartKey = `${inv.id}__${v.color}__${v.size}`;
          if (!_cart.has(cartKey)) {
            _cart.set(cartKey, {
              id: inv.id, cartKey,
              name: `${inv.name} (${v.color} / ${v.size})`,
              price: Number(inv.price), unit: inv.unit || 'pc',
              qty: 1, sizeKey: null, sizeLabel: `${v.color} / ${v.size}`,
              variantColor: v.color, variantSize: v.size
            });
          } else {
            _cart.get(cartKey).qty++;
          }
          _close();
          _refresh(container);
        });
        sizesRow.appendChild(btn);
      });
      sheet.appendChild(sizesRow);
    });
  } else {
    // Color-only picker
    const colorsRow = document.createElement('div');
    colorsRow.style.cssText = 'display:flex;flex-wrap:wrap;gap:8px;';
    inv.variants.forEach(v => {
      const btn = document.createElement('button');
      btn.type = 'button';
      btn.textContent = v.color + (v.qty === 0 ? ' (out)' : '');
      btn.disabled = v.qty === 0;
      btn.className = 'btn btn-secondary';
      btn.style.cssText = 'flex:1;min-width:80px;';
      btn.addEventListener('click', () => {
        const cartKey = `${inv.id}__${v.color}`;
        if (!_cart.has(cartKey)) {
          _cart.set(cartKey, {
            id: inv.id, cartKey,
            name: `${inv.name} (${v.color})`,
            price: Number(inv.price), unit: inv.unit || 'pc',
            qty: 1, sizeKey: null, sizeLabel: v.color,
            variantColor: v.color, variantSize: null
          });
        } else {
          _cart.get(cartKey).qty++;
        }
        _close();
        _refresh(container);
      });
      colorsRow.appendChild(btn);
    });
    sheet.appendChild(colorsRow);
  }

  const cancelBtn = document.createElement('button');
  cancelBtn.type = 'button';
  cancelBtn.textContent = 'Cancel';
  cancelBtn.className = 'btn btn-ghost btn-full';
  cancelBtn.style.marginTop = '16px';
  cancelBtn.addEventListener('click', _close);
  sheet.appendChild(cancelBtn);

  function _close() { document.body.style.overflow = ''; overlay.remove(); }
  overlay.addEventListener('click', e => { if (e.target === overlay) _close(); });
  overlay.appendChild(sheet);
  document.body.appendChild(overlay);
}
```
