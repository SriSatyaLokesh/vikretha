  <action>
Create `scripts/deploy-rules.js` as a standalone Node.js script (no npm dependencies — uses only built-in `fs`, `path`, `child_process`):

```javascript
#!/usr/bin/env node
/**
 * scripts/deploy-rules.js
 * Deploy Firestore security rules to Firebase.
 * Usage: node scripts/deploy-rules.js
 * Requires: Firebase CLI installed globally (npm install -g firebase-tools)
 * and `firebase login` already completed.
 */
'use strict';

const { execSync } = require('child_process');
const fs   = require('fs');
const path = require('path');

const ROOT       = path.resolve(__dirname, '..');
const RULES_FILE = path.join(ROOT, 'firestore.rules');

// ── Preflight checks ──────────────────────────────────────────────────────
if (!fs.existsSync(RULES_FILE)) {
  console.error('\x1b[31m✗ firestore.rules not found at:\x1b[0m', RULES_FILE);
  process.exit(1);
}

// Check Firebase CLI is available
try {
  execSync('firebase --version', { stdio: 'pipe' });
} catch {
  console.error('\x1b[31m✗ Firebase CLI not found.\x1b[0m Run: npm install -g firebase-tools');
  process.exit(1);
}

// ── Deploy ────────────────────────────────────────────────────────────────
console.log('\x1b[36m→ Deploying Firestore rules from:\x1b[0m', RULES_FILE);
const rulesPreview = fs.readFileSync(RULES_FILE, 'utf8').split('\n').slice(0, 5).join('\n');
console.log('\x1b[90m' + rulesPreview + '\n  ...\x1b[0m\n');

try {
  execSync('firebase deploy --only firestore:rules', {
    cwd:   ROOT,
    stdio: 'inherit',
  });
  console.log('\n\x1b[32m✓ Firestore rules deployed successfully.\x1b[0m');
} catch (err) {
  console.error('\n\x1b[31m✗ Deploy failed. Ensure you are logged in:\x1b[0m firebase login');
  process.exit(1);
}
```

Also create the `scripts/` directory. The script uses ANSI colour codes for output (cyan, green, red, grey). `stdio: 'inherit'` streams Firebase CLI output directly to the terminal.
