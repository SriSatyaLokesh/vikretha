  --primary-ring:  rgba(16,185,129,0.25);

  --primary:       #0ea5e9;
  --primary-hover: #0284c7;
  --primary-light: #f0f9ff;
  --primary-ring:  rgba(14,165,233,0.25);
}
[data-theme="violet"] {
  --primary:       #7c3aed;
  --primary-hover: #6d28d9;
  --primary-light: #f5f3ff;
  --primary-ring:  rgba(124,58,237,0.25);
}
[data-theme="rose"] {
  --primary:       #f43f5e;
  --primary-hover: #e11d48;
  --primary-light: #fff1f2;

  --primary-hover: #334155;

  --primary-ring:  rgba(71,85,105,0.25);

   DARK MODE — data-dark="true" attribute overrides

  --bg-surface:       var(--slate-800);

  --text-primary:   var(--slate-100);

  --text-muted:     var(--slate-500);

  --border:        var(--slate-700);
  --border-strong: var(--slate-600);

  --danger-bg:  rgba(239,68,68,0.15);
  --success-bg: rgba(34,197,94,0.12);
  --warning-bg: rgba(245,158,11,0.12);
  --info-bg:    rgba(14,165,233,0.12);

  --shadow-xs: 0 1px 2px rgba(0,0,0,0.24);
  --shadow-sm: 0 1px 3px rgba(0,0,0,0.28),0 1px 2px rgba(0,0,0,0.20);
  --shadow-md: 0 4px 8px rgba(0,0,0,0.32),0 2px 4px rgba(0,0,0,0.20);

  --shadow-xl: 0 20px 40px rgba(0,0,0,0.44),0 8px 16px rgba(0,0,0,0.24);
