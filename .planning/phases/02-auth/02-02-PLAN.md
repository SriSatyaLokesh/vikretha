```javascript
rules_version = '2';
// =============================================================================
// Vikretha — Firestore Security Rules
// =============================================================================
// SETUP INSTRUCTIONS:
//   1. Open Firebase Console → Firestore Database → Rules tab
//   2. Copy-paste the rules below, replacing any existing content
//   3. Click "Publish"
//
// HOW IT WORKS:
//   - All data is denied by default (deny-all base rule)
//   - Only email addresses listed in /shops/{shopId}/config.authorized_emails
//     can read or write any document under /shops/{shopId}/
//   - The first "Create Account" bootstraps this config doc with the owner's email
//
// ADDING STAFF:
//   In Firebase Console → Firestore → shops/{your_shop_id}/config
//   Add the staff email to the authorized_emails array and save.
//   Then have them create a Firebase Auth account via the app's "Create Account" form.
// =============================================================================

service cloud.firestore {
  match /databases/{database}/documents {

    // Deny all reads and writes by default
    match /{document=**} {
      allow read, write: if false;
    }

    // Helper function: check if the authenticated user's email
    // is in this shop's authorized_emails list
    function isAuthorized(shopId) {
      let config = get(/databases/$(database)/documents/shops/$(shopId)/config);
      return request.auth != null
        && request.auth.token.email != null
        && request.auth.token.email in config.data.authorized_emails;
    }

    // Shop data: all subcollections accessible to authorized emails only
    match /shops/{shopId}/{document=**} {
      allow read: if isAuthorized(shopId);
      allow write: if isAuthorized(shopId);
    }

    // Bootstrap: allow authenticated user to CREATE config doc if it doesn't exist.
    // Covers the very first "Create Account" before authorized_emails is populated.
    // Security: user can only bootstrap with their OWN email in authorized_emails.
    match /shops/{shopId}/config {
      allow create: if request.auth != null
        && request.auth.token.email != null
        && request.auth.token.email in request.resource.data.authorized_emails;
    }
  }
}
```

```javascript
/**
 * modules/settings.js — Settings Screen
 * Phase 2: Sign-out + current user display.
 * Phase 8 will add: staff email management (authorized_emails), shop profile editing.
 */
import { auth } from '../lib/firebase-init.js';
import { signOut } from 'https://www.gstatic.com/firebasejs/10.12.0/firebase-auth.js';
import { SHOP_NAME, SHOP_ID } from '../shop.config.js';

export function render(container) {
  const user = auth.currentUser;
  const email = user?.email || '—';

  container.innerHTML = `
    <div style="padding-bottom:16px;">
      <div class="card" style="margin-bottom:16px;">
        <p style="font-size:0.75rem;color:var(--text-muted);margin-bottom:4px;">SIGNED IN AS</p>
        <p style="font-weight:600;word-break:break-all;">${escapeHtml(email)}</p>
        <p style="font-size:0.75rem;color:var(--text-muted);margin-top:8px;">Shop ID: ${escapeHtml(SHOP_ID)}</p>
      </div>

      <div class="card" style="margin-bottom:16px;">
        <p style="font-size:0.75rem;color:var(--text-muted);margin-bottom:12px;
                  text-transform:uppercase;letter-spacing:0.05em;">Account</p>
        <button id="sign-out-btn" class="btn btn-full"
                style="justify-content:flex-start;color:var(--danger);border:1px solid var(--border);">
          Sign Out
        </button>
      </div>

      <div class="card">
        <p style="font-size:0.75rem;color:var(--text-muted);margin-bottom:12px;
                  text-transform:uppercase;letter-spacing:0.05em;">Coming Soon</p>
        <p style="font-size:0.875rem;color:var(--text-secondary);">
          Staff access management, shop profile editing, and more in a future update.
        </p>
      </div>
    </div>
  `;

  document.getElementById('sign-out-btn').addEventListener('click', handleSignOut);
}

async function handleSignOut() {
  const btn = document.getElementById('sign-out-btn');
  btn.disabled = true;
  btn.textContent = 'Signing out…';
  try {
    await signOut(auth);
    // onAuthStateChanged in app.js fires with null → showLoginScreen()
  } catch (err) {
    btn.disabled = false;
    btn.textContent = 'Sign Out';
    console.error('[Vikretha] Sign out failed:', err);
  }
}

function escapeHtml(str) {
  return String(str).replace(/&/g, '&amp;').replace(/"/g, '&quot;').replace(/</g, '&lt;');
}
```
