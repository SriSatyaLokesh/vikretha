```javascript
rules_version = '2';
// =============================================================================
// Vikretha — Firestore Security Rules
// =============================================================================
// SETUP INSTRUCTIONS:
//   1. Open Firebase Console → Firestore Database → Rules tab
//   2. Copy-paste the rules below, replacing any existing content
//   3. Click "Publish"
//
// HOW IT WORKS:
//   - All data is denied by default (deny-all base rule)
//   - Only phone numbers listed in /shops/{shopId}/config.authorized_phones
//     can read or write any document under /shops/{shopId}/
//   - The first sign-in bootstraps this config doc with the owner's phone
//
// ADDING STAFF:
//   In Firebase Console → Firestore → shops/{your_shop_id}/config
//   Add the staff phone number (E.164 format: +919876543210) to the
//   authorized_phones array and save.
// =============================================================================

service cloud.firestore {
  match /databases/{database}/documents {

    // Deny all reads and writes by default
    match /{document=**} {
      allow read, write: if false;
    }

    // Helper function: check if the authenticated user's phone number
    // is in this shop's authorized_phones list
    function isAuthorized(shopId) {
      let config = get(/databases/$(database)/documents/shops/$(shopId)/config);
      return request.auth != null
        && request.auth.token.phone_number != null
        && request.auth.token.phone_number in config.data.authorized_phones;
    }

    // Shop data: all subcollections accessible to authorized phones only
    match /shops/{shopId}/{document=**} {
      allow read: if isAuthorized(shopId);
      allow write: if isAuthorized(shopId);
    }
  }
}
```

Note on bootstrapping: The very first login (shop owner's first sign-in) calls
`bootstrapShopConfig()` in modules/auth.js which creates the config doc BEFORE
these rules would reject the write. This works because the config doc doesn't
exist yet — Firestore allows the write to proceed through the `if false` rule
only if the caller is authenticated. Wait — actually this is a chicken-and-egg:
the deny-all rule blocks the bootstrap write.

RESOLUTION: Add a one-time bootstrap rule that allows authenticated users to
create a non-existent config doc. Modify the rules as follows:

```javascript
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Deny all by default
    match /{document=**} {
      allow read, write: if false;
    }

    function isAuthorized(shopId) {
      let config = get(/databases/$(database)/documents/shops/$(shopId)/config);
      return request.auth != null
        && request.auth.token.phone_number != null
        && request.auth.token.phone_number in config.data.authorized_phones;
    }

    match /shops/{shopId}/{document=**} {
      allow read: if isAuthorized(shopId);
      allow write: if isAuthorized(shopId);
    }

    // Bootstrap: allow authenticated user to CREATE config doc if it doesn't exist
    // This covers the first-ever sign-in before authorized_phones is populated
    match /shops/{shopId}/config {
      allow create: if request.auth != null
        && request.auth.token.phone_number != null
        && request.auth.token.phone_number in request.resource.data.authorized_phones;
    }
  }
}
```

Use this final version (with bootstrap rule) in the actual firestore.rules file.

The bootstrap rule ensures:
- Only authenticated users can create the config doc
- They can only create it with their OWN phone in authorized_phones (can't spoof)
- Once created, the isAuthorized function takes over for all subsequent writes

/**
 * modules/settings.js — Settings Screen
 * Phase 2: Sign-out only.
 * Phase 8 will add: staff management (authorized_phones), shop profile editing.

import { auth } from '../lib/firebase-init.js';
import { signOut } from 'https://www.gstatic.com/firebasejs/10.12.0/firebase-auth.js';
import { SHOP_NAME, SHOP_ID } from '../shop.config.js';

export function render(container) {
  const user = auth.currentUser;
  const phone = user?.phoneNumber || '—';

  container.innerHTML = `
    <div style="padding-bottom:16px;">
      <div class="card" style="margin-bottom:16px;">
        <p style="font-size:0.75rem;color:var(--text-muted);margin-bottom:4px;">SIGNED IN AS</p>
        <p style="font-weight:600;">${phone}</p>
        <p style="font-size:0.75rem;color:var(--text-muted);margin-top:8px;">Shop ID: ${SHOP_ID}</p>
      </div>

      <div class="card" style="margin-bottom:16px;">
        <p style="font-size:0.75rem;color:var(--text-muted);margin-bottom:12px;
                  text-transform:uppercase;letter-spacing:0.05em;">Account</p>
        <button id="sign-out-btn" class="btn btn-full"
                style="justify-content:flex-start;color:var(--danger);border:1px solid var(--border);">
          Sign Out
        </button>
      </div>

      <div class="card">
        <p style="font-size:0.75rem;color:var(--text-muted);margin-bottom:12px;
                  text-transform:uppercase;letter-spacing:0.05em;">Coming Soon</p>
        <p style="font-size:0.875rem;color:var(--text-secondary);">
          Staff management, shop profile editing, and more in a future update.
        </p>
      </div>
    </div>
  `;

  document.getElementById('sign-out-btn').addEventListener('click', handleSignOut);
}

async function handleSignOut() {
  const btn = document.getElementById('sign-out-btn');
  btn.disabled = true;
  btn.textContent = 'Signing out…';
  try {
    await signOut(auth);
    // onAuthStateChanged in app.js fires with null → showLoginScreen()
    // No manual navigation needed
  } catch (err) {
    btn.disabled = false;
    btn.textContent = 'Sign Out';
    console.error('[Vikretha] Sign out failed:', err);
  }
}

| Threat ID | Category | Component | Disposition | Mitigation Plan |

| T-02-06 | Elevation of Privilege | firestore.rules bootstrap rule | mitigate | Bootstrap `allow create` rule requires `request.auth.token.phone_number in request.resource.data.authorized_phones` — user can only bootstrap with their own phone, preventing spoofing other users |
| T-02-07 | Elevation of Privilege | Cross-shop data access | mitigate | isAuthorized(shopId) reads the specific shop's config — a user authorized for shop_A cannot access shop_B even with a valid Firebase token |
| T-02-08 | Tampering | authorized_phones modification | mitigate | Writing to /shops/{shopId}/config requires isAuthorized check — only existing authorized phones can add new ones (except bootstrap case) |
