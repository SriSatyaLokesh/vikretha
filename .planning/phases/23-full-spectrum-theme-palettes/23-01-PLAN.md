```css

  /* Accent */
  --primary: HEX; --primary-hover: HEX; --primary-light: HEX; --primary-ring: rgba(...);
  /* Light mode backgrounds */
  --bg-app: HEX; --bg-surface: HEX; --bg-elevated: HEX;
  --bg-sidebar: HEX; --bg-sidebar-hover: HEX;
  /* Light mode text/border */
  --text-primary: HEX; --border: HEX;
}
[data-theme="THEME-ID"][data-dark="true"] {
  /* Dark mode backgrounds */
  --bg-app: HEX; --bg-surface: HEX; --bg-elevated: HEX;
  --bg-sidebar: HEX; --bg-sidebar-hover: HEX;
  /* Dark mode text/border */
  --text-primary: HEX; --text-secondary: HEX; --border: HEX; --border-strong: HEX;
  /* Primary tint on dark */
  --primary-light: rgba(...);
}
```
