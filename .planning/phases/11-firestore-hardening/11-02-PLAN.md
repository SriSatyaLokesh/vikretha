```js
async function _fetchAndRenderStats(container) {
  const now = new Date();
  const fmt = v => CURRENCY + v.toLocaleString(LOCALE, { minimumFractionDigits: 2 });

  // ── Read 7 daily_summary docs (7 reads instead of hundreds) ──
  const dayMap = {};
  let todayRev = 0, todayCt = 0, weekRev = 0, weekCt = 0;
  const todayKey = now.toISOString().slice(0, 10);

  const dayPromises = [];
  for (let i = 6; i >= 0; i--) {
    const d = new Date(now);
    d.setDate(d.getDate() - i);
    const key = d.toISOString().slice(0, 10);
    dayMap[key] = { revenue: 0, count: 0 };
    dayPromises.push(
      getDoc(doc(db, 'shops', SHOP_ID, 'daily_summary', key))
        .then(snap => {
          if (snap.exists()) {
            const data = snap.data();
            dayMap[key].revenue = data.revenue || 0;
            dayMap[key].count   = data.count || 0;
          }
        })
    );
  }
  await Promise.all(dayPromises);

  // Compute today + week from daily docs
  for (const [key, val] of Object.entries(dayMap)) {
    weekRev += val.revenue;
    weekCt  += val.count;
    if (key === todayKey) {
      todayRev = val.revenue;
      todayCt  = val.count;
    }
  }

  // ── Read 1 monthly_summary doc ──
  const monthKey = todayKey.slice(0, 7); // "YYYY-MM"
  let monthRev = 0, monthCt = 0;
  try {
    const monthSnap = await getDoc(doc(db, 'shops', SHOP_ID, 'monthly_summary', monthKey));
    if (monthSnap.exists()) {
      monthRev = monthSnap.data().revenue || 0;
      monthCt  = monthSnap.data().count || 0;
    }
  } catch (e) { console.warn('[Dashboard] monthly_summary read failed', e); }

  // ── Render ──
  _setSlot(container, 'dash-today-rev',  fmt(todayRev));
  _setSlot(container, 'dash-today-ct',   todayCt + ' sale' + (todayCt !== 1 ? 's' : ''));
  _setSlot(container, 'dash-week-rev',   fmt(weekRev));
  _setSlot(container, 'dash-week-ct',    weekCt + ' sale' + (weekCt !== 1 ? 's' : ''));
  _setSlot(container, 'dash-month-rev',  fmt(monthRev));
  _setSlot(container, 'dash-month-ct',   monthCt + ' sale' + (monthCt !== 1 ? 's' : ''));
  _renderBarChart(container, dayMap);
}
```

**Part B — Create scripts/backfill-summaries.js:**

Create a Node.js script that reads all existing sales and populates daily_summary + monthly_summary docs:

```js
#!/usr/bin/env node
/**
 * scripts/backfill-summaries.js
 * One-time migration: reads all sales and creates daily_summary + monthly_summary docs.
 * Run: node scripts/backfill-summaries.js <email> <password>
 * Requires: firebase-admin or firebase client SDK (uses client SDK for simplicity)
 */
```

The script should:
1. Accept email + password as CLI args (for Firebase Auth sign-in)
2. Import Firebase client SDK (from node_modules or use dynamic import from CDN — for simplicity, use `firebase` npm package or just document that the user should run it in browser console)
3. Query ALL documents from `shops/{SHOP_ID}/sales`
4. For each sale, extract `timestamp` and `total`
5. Aggregate into daily and monthly maps: `{ [YYYY-MM-DD]: { count, revenue }, ... }`
6. Write each aggregated doc using `setDoc` (not increment — this is a full overwrite for backfill)
7. Print progress: "Processed X sales → Y daily docs, Z monthly docs"

**IMPORTANT:** Since this project has no build step and no node_modules, make the backfill script a browser-compatible ES module that can be pasted into the browser console, OR make it a Node.js script that uses the firebase npm package.

Best approach: Create it as a **browser console script** with inline Firebase config (matches the project's no-build philosophy). The user pastes it in DevTools console while logged in:

```js
// Paste in browser console while logged in at the app
(async () => {
  const { db } = await import('./lib/firebase-init.js');
  const { collection, getDocs, doc, setDoc } = await import('https://www.gstatic.com/firebasejs/10.12.0/firebase-firestore.js');
  const { SHOP_ID } = await import('./shop.config.js');

  console.log('Fetching all sales...');
  const snap = await getDocs(collection(db, 'shops', SHOP_ID, 'sales'));
  const daily = {}, monthly = {};

  snap.forEach(s => {
    const d = s.data();
    if (!d.timestamp) return;
    const ts = d.timestamp.toDate();
    const dayKey = ts.toISOString().slice(0, 10);
    const monthKey = dayKey.slice(0, 7);
    const total = d.total || 0;

    if (!daily[dayKey]) daily[dayKey] = { date: dayKey, count: 0, revenue: 0 };
    daily[dayKey].count += 1;
    daily[dayKey].revenue += total;

    if (!monthly[monthKey]) monthly[monthKey] = { month: monthKey, count: 0, revenue: 0 };
    monthly[monthKey].count += 1;
    monthly[monthKey].revenue += total;
  });

  console.log(`Processed ${snap.size} sales → ${Object.keys(daily).length} daily, ${Object.keys(monthly).length} monthly docs`);

  for (const [key, data] of Object.entries(daily)) {
    await setDoc(doc(db, 'shops', SHOP_ID, 'daily_summary', key), data);
  }
  for (const [key, data] of Object.entries(monthly)) {
    await setDoc(doc(db, 'shops', SHOP_ID, 'monthly_summary', key), data);
  }

  console.log('✓ Backfill complete!');
})();
```

Save this as `scripts/backfill-summaries.js` with a comment header explaining usage:
